##### 20、实现普通对象的可迭代？
```js
var obj = {
  name: 'H2O',
  age: 100,
  address: 'xxxxx'
}

obj[Symbol.iterator] = function () {
  let index = 0
  let keys = Object.keys(this)
  let self = this

  return {
    next() {
      if (index < keys.length) {
        return {
          value: self[keys[index++]],
          done: false
        }
      } else {
        return {
          value: undefined,
          done: true
        }
      }
    }
  }
}

// 测试：
for (let item of obj) {
  console.log(item);
}
```
